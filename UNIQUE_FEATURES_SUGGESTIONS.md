#### **Step 2: Implement K-Means Clustering (2 hours)**
```typescript
// Simple K-means implementation (no libraries needed!)

function kMeansClustering(
  customers: CustomerFeatures[], 
  k: number = 4
): Cluster[] {
  // Normalize features to 0-1 scale
  const normalized = normalizeFeatures(customers);
  
  // Initialize random centroids
  let centroids = initializeCentroids(normalized, k);
  
  // Iterate until convergence
  for (let iter = 0; iter < 100; iter++) {
    // Assign customers to nearest centroid
    const clusters = assignToClusters(normalized, centroids);
    
    // Update centroids
    const newCentroids = updateCentroids(clusters);
    
    // Check convergence
    if (centroidsEqual(centroids, newCentroids)) break;
    centroids = newCentroids;
  }
  
  return clusters;
}

function assignToClusters(
  customers: CustomerFeatures[], 
  centroids: number[][]
): Cluster[] {
  const clusters: Cluster[] = Array(centroids.length).fill([]);
  
  customers.forEach(customer => {
    // Find nearest centroid using Euclidean distance
    let minDist = Infinity;
    let clusterIndex = 0;
    
    centroids.forEach((centroid, i) => {
      const dist = euclideanDistance(customer, centroid);
      if (dist < minDist) {
        minDist = dist;
        clusterIndex = i;
      }
    });
    
    clusters[clusterIndex].push(customer);
  });
  
  return clusters;
}
```
